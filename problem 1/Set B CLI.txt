For R1-ISP

enable
configure terminal

! ---------------- Housekeeping ----------------
hostname R1-ISP
enable secret csepstu
banner motd "This is lab router R1-ISP. Authorized access only."
no ip domain-lookup

line console 0
 password csepstu
 login
line vty 0 4
 password csepstu
 login

! ---------------- LAN Interface ----------------
interface FastEthernet0/0
 description R1-ISP_LAN
 ip address 172.169.254.62 255.255.255.192
 no shutdown

! ---------------- WAN to R2-Central ----------------
interface Serial0/0/0
 description Link_to_R2-Central
 ip address 172.169.254.129 255.255.255.252
 clock rate 64000
 no shutdown

! ---------------- WAN to R3-ISP ----------------
interface Serial0/0/1
 description Link_to_R3-ISP
 ip address 172.169.254.133 255.255.255.252
 clock rate 64000
 no shutdown

! ---------------- Static Routes ----------------
! R2-Central LAN 1
ip route 172.169.254.64 255.255.255.192 172.169.254.130

! R2-Central LAN 2
ip route 172.169.254.96 255.255.255.224 172.169.254.130

! R3-ISP LAN
ip route 172.169.254.32 255.255.255.224 172.169.254.134

end
